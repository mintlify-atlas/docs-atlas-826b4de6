## Overview

The Trading Data Gateway (TDG) provides access to all trading operations including order management, position tracking, account data, and transfers. All TDG endpoints require authentication.

```typescript
import { TDG } from '@grvt/client';

const tdg = new TDG({
  host: 'https://api.grvt.io',
  version: 'v1'
});
```

<Note>
Make sure you've configured your API credentials and have the necessary permissions before executing trades.
</Note>


## Order Management

### Create Limit Orders

```typescript
// Buy limit order
const buyOrder = await tdg.createOrder({
  sub_account_id: 'YOUR_SUB_ACCOUNT_ID',
  instrument: 'BTC_USDT_Perp',
  is_market: false,
  size: '0.1', // Buy 0.1 BTC
  limit_price: '50000.00',
  time_in_force: 'GTC', // Good till cancelled
  post_only: true // Ensure maker order
});

console.log('Order ID:', buyOrder.result.order_id);
console.log('Status:', buyOrder.result.metadata.status);

// Sell limit order
const sellOrder = await tdg.createOrder({
  sub_account_id: 'YOUR_SUB_ACCOUNT_ID',
  instrument: 'BTC_USDT_Perp',
  is_market: false,
  size: '-0.1', // Sell 0.1 BTC (negative size)
  limit_price: '52000.00',
  time_in_force: 'GTC'
});
```

### Create Orders with Stop Loss and Take Profit

```typescript
// Open a long position with SL/TP
const entryOrder = await tdg.createOrder({
  sub_account_id: 'YOUR_SUB_ACCOUNT_ID',
  instrument: 'BTC_USDT_Perp',
  is_market: true,
  size: '0.5',
  time_in_force: 'IOC'
});

const entryPrice = parseFloat(entryOrder.result.metadata.avg_fill_price);

// Set stop loss 2% below entry
const stopLoss = await tdg.createOrder({
  sub_account_id: 'YOUR_SUB_ACCOUNT_ID',
  instrument: 'BTC_USDT_Perp',
  is_market: false,
  size: '-0.5',
  trigger_price: (entryPrice * 0.98).toFixed(2),
  time_in_force: 'GTC',
  order_type: 'STOP_LOSS'
});

// Set take profit 5% above entry
const takeProfit = await tdg.createOrder({
  sub_account_id: 'YOUR_SUB_ACCOUNT_ID',
  instrument: 'BTC_USDT_Perp',
  is_market: false,
  size: '-0.5',
  limit_price: (entryPrice * 1.05).toFixed(2),
  time_in_force: 'GTC'
});

console.log('Position opened with SL/TP');
console.log('Stop Loss ID:', stopLoss.result.order_id);
console.log('Take Profit ID:', takeProfit.result.order_id);
```

### Close Position

```typescript
async function closePosition(subAccountId: string, instrument: string) {
  // Get current position
  const positions = await tdg.positions({
    sub_account_id: subAccountId,
    instrument
  });

  const position = positions.result.find(p => p.instrument === instrument);
  if (!position) {
    console.log('No position found');
    return;
  }

  const positionSize = parseFloat(position.size);
  if (positionSize === 0) {
    console.log('Position already closed');
    return;
  }

  // Close with market order (opposite side)
  const closeOrder = await tdg.createOrder({
    sub_account_id: subAccountId,
    instrument,
    is_market: true,
    size: (-positionSize).toString(), // Negative to close
    time_in_force: 'IOC'
  });

  console.log('Position closed');
  console.log('Close price:', closeOrder.result.metadata.avg_fill_price);
  console.log('Realized PnL:', closeOrder.result.metadata.realized_pnl);
}

await closePosition('YOUR_SUB_ACCOUNT_ID', 'BTC_USDT_Perp');
```

## Advanced Trading Strategies

### Grid Trading Bot

```typescript
class GridTradingBot {
  constructor(
    private tdg: TDG,
    private subAccountId: string,
    private instrument: string,
    private gridLevels: number,
    private gridSpacing: number,
    private orderSize: string
  ) {}

  async start(basePrice: number) {
    console.log(`Starting grid bot at base price: ${basePrice}`);
    
    const orders = [];
    
    // Create buy orders below base price
    for (let i = 1; i <= this.gridLevels; i++) {
      const buyPrice = (basePrice * (1 - i * this.gridSpacing)).toFixed(2);
      orders.push({
        instrument: this.instrument,
        is_market: false,
        size: this.orderSize,
        limit_price: buyPrice,
        time_in_force: 'GTC'
      });
    }
    
    // Create sell orders above base price
    for (let i = 1; i <= this.gridLevels; i++) {
      const sellPrice = (basePrice * (1 + i * this.gridSpacing)).toFixed(2);
      orders.push({
        instrument: this.instrument,
        is_market: false,
        size: `-${this.orderSize}`,
        limit_price: sellPrice,
        time_in_force: 'GTC'
      });
    }
    
    const result = await this.tdg.createBulkOrders({
      sub_account_id: this.subAccountId,
      orders
    });
    
    console.log(`Grid bot started with ${result.result.length} orders`);
    return result.result;
  }

  async stop() {
    await this.tdg.cancelAllOrders({
      sub_account_id: this.subAccountId,
      instrument: this.instrument
    });
    console.log('Grid bot stopped');
  }
}

const gridBot = new GridTradingBot(
  tdg,
  'YOUR_SUB_ACCOUNT_ID',
  'BTC_USDT_Perp',
  5, // 5 levels on each side
  0.01, // 1% spacing
  '0.01' // 0.01 BTC per order
);

await gridBot.start(50000); // Start at $50,000
```

### TWAP Order Execution

```typescript
async function executeTWAP(
  tdg: TDG,
  subAccountId: string,
  instrument: string,
  totalSize: number,
  durationMinutes: number,
  numOrders: number
) {
  const sizePerOrder = totalSize / numOrders;
  const intervalMs = (durationMinutes * 60 * 1000) / numOrders;
  
  console.log(`Executing TWAP: ${totalSize} over ${durationMinutes} minutes`);
  console.log(`${numOrders} orders of ${sizePerOrder} each`);
  
  for (let i = 0; i < numOrders; i++) {
    await new Promise(resolve => setTimeout(resolve, intervalMs));
    
    try {
      const order = await tdg.createOrder({
        sub_account_id: subAccountId,
        instrument,
        is_market: true,
        size: sizePerOrder.toString(),
        time_in_force: 'IOC'
      });
      
      console.log(`Order ${i + 1}/${numOrders} filled @ ${order.result.metadata.avg_fill_price}`);
    } catch (error) {
      console.error(`Order ${i + 1} failed:`, error.message);
    }
  }
  
  console.log('TWAP execution complete');
}

await executeTWAP(
  tdg,
  'YOUR_SUB_ACCOUNT_ID',
  'BTC_USDT_Perp',
  1.0, // Buy 1 BTC
  30, // Over 30 minutes
  10 // Split into 10 orders
);
```
