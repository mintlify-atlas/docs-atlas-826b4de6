## Overview

The GRVT WebSocket API provides real-time streaming data for market feeds and account updates. WebSocket connections are more efficient than REST polling for frequently updated data.

```typescript
import { WS, EStream } from '@grvt/client';

const ws = new WS({
  url: 'wss://ws.grvt.io',
  reconnectStrategy: (retries) => {
    // Exponential backoff with max 30s
    return Math.min(1000 * Math.pow(2, retries), 30000);
  },
  timeout: 30000
});
```

## Connection Management

### Connect and Disconnect

```typescript
// Connect to WebSocket
ws.connect();

// Wait for connection to be ready
await ws.ready();
console.log('WebSocket connected');

// Disconnect when done
ws.disconnect();
```

### Connection Event Handlers

```typescript
// Handle connection events
ws.onConnect(() => {
  console.log('WebSocket connected');
});

ws.onClose((event) => {
  console.log('WebSocket closed:', event.code, event.reason);
});

ws.onError((error) => {
  console.error('WebSocket error:', error);
});

// Connect
ws.connect();
```

### Multi-Instrument Price Monitor

```typescript
class MultiInstrumentMonitor {
  private subscriptions: string[] = [];
  private prices: Map<string, string> = new Map();

  constructor(private ws: WS, private instruments: string[]) {}

  start() {
    this.subscriptions = this.ws.subscribes(
      ...this.instruments.map(instrument => ({
        stream: EStream.MINI_SNAP,
        params: { instrument, rate: 1000 },
        onData: (mini: any) => {
          this.prices.set(instrument, mini.last_price);
          this.displayPrices();
        }
      }))
    );
  }

  stop() {
    this.ws.unsubscribe(...this.subscriptions);
    this.subscriptions = [];
  }

  private displayPrices() {
    console.clear();
    console.log('Real-time Prices:\n');
    this.prices.forEach((price, instrument) => {
      console.log(`${instrument.padEnd(20)} $${price}`);
    });
  }
}

const monitor = new MultiInstrumentMonitor(ws, [
  'BTC_USDT_Perp',
  'ETH_USDT_Perp',
  'SOL_USDT_Perp',
  'AVAX_USDT_Perp'
]);

monitor.start();

// Stop after 1 minute
setTimeout(() => monitor.stop(), 60000);
```

## Advanced Patterns

### Order Book Aggregator

```typescript
class OrderBookAggregator {
  private bids: Map<string, string> = new Map();
  private asks: Map<string, string> = new Map();

  constructor(private ws: WS, private instrument: string) {}

  start() {
    // Subscribe to book snapshots first
    this.ws.subscribe({
      stream: EStream.RPI_BOOK_SNAP,
      params: {
        instrument: this.instrument,
        depth: 20,
        rate: 1000
      },
      onData: (book) => {
        // Initialize with snapshot
        this.bids.clear();
        this.asks.clear();
        
        book.bids?.forEach(bid => {
          this.bids.set(bid.price, bid.size);
        });
        book.asks?.forEach(ask => {
          this.asks.set(ask.price, ask.size);
        });
      }
    });

    // Then subscribe to deltas
    this.ws.subscribe({
      stream: EStream.RPI_BOOK_DELTA,
      params: {
        instrument: this.instrument,
        rate: 100
      },
      onData: (delta) => {
        // Apply incremental updates
        delta.bids?.forEach(bid => {
          if (parseFloat(bid.size) === 0) {
            this.bids.delete(bid.price);
          } else {
            this.bids.set(bid.price, bid.size);
          }
        });
        
        delta.asks?.forEach(ask => {
          if (parseFloat(ask.size) === 0) {
            this.asks.delete(ask.price);
          } else {
            this.asks.set(ask.price, ask.size);
          }
        });
      }
    });
  }

  getBestBid(): [string, string] | undefined {
    const prices = Array.from(this.bids.keys()).sort((a, b) => parseFloat(b) - parseFloat(a));
    const bestPrice = prices[0];
    return bestPrice ? [bestPrice, this.bids.get(bestPrice)!] : undefined;
  }

  getBestAsk(): [string, string] | undefined {
    const prices = Array.from(this.asks.keys()).sort((a, b) => parseFloat(a) - parseFloat(b));
    const bestPrice = prices[0];
    return bestPrice ? [bestPrice, this.asks.get(bestPrice)!] : undefined;
  }

  getSpread(): number {
    const bid = this.getBestBid();
    const ask = this.getBestAsk();
    if (!bid || !ask) return 0;
    return parseFloat(ask[0]) - parseFloat(bid[0]);
  }
}

const bookAgg = new OrderBookAggregator(ws, 'BTC_USDT_Perp');
bookAgg.start();

setInterval(() => {
  const bid = bookAgg.getBestBid();
  const ask = bookAgg.getBestAsk();
  const spread = bookAgg.getSpread();
  
  console.log(`Best Bid: ${bid?.[0]} (${bid?.[1]})`);
  console.log(`Best Ask: ${ask?.[0]} (${ask?.[1]})`);
  console.log(`Spread: ${spread}`);
}, 5000);
```

### Real-time PnL Monitor

```typescript
class PnLMonitor {
  private currentPrice: number = 0;
  private positions: any[] = [];

  constructor(
    private ws: WS,
    private subAccountId: string,
    private instrument: string
  ) {}

  start() {
    // Subscribe to position updates
    this.ws.subscribe({
      stream: EStream.POSITION,
      params: {
        sub_account_id: this.subAccountId,
        instrument: this.instrument
      },
      onData: (positions) => {
        this.positions = positions;
        this.calculatePnL();
      }
    });

    // Subscribe to price updates
    this.ws.subscribe({
      stream: EStream.MINI_SNAP,
      params: {
        instrument: this.instrument,
        rate: 500
      },
      onData: (mini) => {
        this.currentPrice = parseFloat(mini.last_price);
        this.calculatePnL();
      }
    });
  }

  private calculatePnL() {
    if (!this.currentPrice || !this.positions.length) return;

    this.positions.forEach(position => {
      const size = parseFloat(position.size);
      const entryPrice = parseFloat(position.avg_entry_price);
      const unrealizedPnL = (this.currentPrice - entryPrice) * size;
      const pnlPercent = ((this.currentPrice - entryPrice) / entryPrice) * 100;

      console.log(`\n${position.instrument} Position:`);
      console.log(`  Size: ${size}`);
      console.log(`  Entry: $${entryPrice}`);
      console.log(`  Current: $${this.currentPrice}`);
      console.log(`  Unrealized PnL: $${unrealizedPnL.toFixed(2)} (${pnlPercent.toFixed(2)}%)`);
    });
  }
}

const pnlMonitor = new PnLMonitor(ws, 'YOUR_SUB_ACCOUNT_ID', 'BTC_USDT_Perp');
pnlMonitor.start();
```

## Error Handling

```typescript
const subscription = ws.subscribe({
  stream: EStream.TICKER_SNAP,
  params: {
    instrument: 'BTC_USDT_Perp',
    rate: 1000
  },
  onData: (ticker) => {
    try {
      // Process ticker data
      const price = parseFloat(ticker.last_price);
      console.log('Price:', price);
    } catch (error) {
      console.error('Error processing ticker:', error);
    }
  },
  onError: (error) => {
    console.error('Subscription error:', error.message);
    
    // Attempt to resubscribe
    if (error.code === 1003) {
      console.log('Malformed request, not resubscribing');
    } else {
      console.log('Resubscribing...');
      // Resubscription handled automatically by reconnect strategy
    }
  }
});
```

## Best Practices

<Note>
1. **Use batch subscriptions** when subscribing to multiple streams
2. **Implement exponential backoff** in reconnect strategy
3. **Handle snapshots before deltas** for order book data
4. **Clean up subscriptions** when no longer needed
5. **Monitor connection health** with heartbeat tracking
</Note>

### Resource Cleanup

```typescript
class WebSocketManager {
  private ws: WS;
  private subscriptions: string[] = [];

  constructor(url: string) {
    this.ws = new WS({
      url,
      reconnectStrategy: (retries) => Math.min(1000 * Math.pow(2, retries), 30000)
    });
  }

  subscribe(config: any) {
    const sub = this.ws.subscribe(config);
    this.subscriptions.push(sub);
    return sub;
  }

  cleanup() {
    // Unsubscribe from all
    this.ws.unsubscribe(...this.subscriptions);
    this.subscriptions = [];
    
    // Disconnect
    this.ws.disconnect();
  }
}

const manager = new WebSocketManager('wss://ws.grvt.io');

// Use the manager...

// Clean up when done
process.on('SIGINT', () => {
  console.log('Cleaning up...');
  manager.cleanup();
  process.exit();
});
```
